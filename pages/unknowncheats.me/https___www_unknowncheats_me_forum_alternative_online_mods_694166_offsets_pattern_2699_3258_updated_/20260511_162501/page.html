uintptr_t FindPattern(const char* pattern, const char* mask, uintptr_t baseAddress, size_t size)
{
	for (size_t i = 0; i < size; i++)
	{
		bool found = true;
		for (size_t j = 0; mask[j] != '\0'; j++)
		{
			if (mask[j] == 'x' && pattern[j] != *(char*)(baseAddress + i + j))
			{
				found = false;
				break;
			}
		}
		if (found)
		{
			return baseAddress + i;
		}
	}
	return 0;

uintptr_t FindPattern(const char* pattern, const char* mask, uintptr_t baseAddress, size_t size)
{
	for (size_t i = 0; i < size; i++)
	{
		bool found = true;
		for (size_t j = 0; mask[j] != '\0'; j++)
		{
			if (mask[j] == 'x' && pattern[j] != *(char*)(baseAddress + i + j))
			{
				found = false;
				break;
			}
		}
		if (found)
		{
			return baseAddress + i;
		}
	}
	return 0;
}
